BEGIN

    SET RECIPES_FILE_PATH
    SET INGREDIENTS_MAPPING_FILE_PATH
    SET MEASUREMENTS_MAPPING_FILE_PATH
    SET BILKA_PRICES_FILE_PATH
    SET NETTO_PRICES_FILE_PATH
    SET FOETEX_PRICES_FILE_PATH
    SET REMA_PRICES_FILE_PATH


    FUNCTION GetInputs()

        REPEAT
            DISPLAY "Minimum price per recipe (DKK): "
            READ budgetMin

            IF budgetMin is not a number THEN
                DISPLAY "Please choose a number"
                WAIT for user
            ELSE IF budgetMin < 0 THEN
                DISPLAY "Please choose a number above 0"
                WAIT for user
            ELSE
                EXIT LOOP
            END IF
        UNTIL valid input


        REPEAT
            DISPLAY "Maximum price per recipe (DKK): "
            READ budgetMax

            IF budgetMax is not a number THEN
                DISPLAY "Please choose a number"
                WAIT for user
            ELSE IF budgetMax < budgetMin THEN
                DISPLAY "Max price can not be lower than minimum price"
                WAIT for user
            ELSE IF budgetMax <= 0 THEN
                DISPLAY "Please choose a number above 0"
                WAIT for user
            ELSE
                EXIT LOOP
            END IF
        UNTIL valid input


        REPEAT
            DISPLAY "Amount of recipes: "
            READ recipesAmount

            IF recipesAmount is not a whole number THEN
                DISPLAY "Please choose a whole number"
                WAIT for user
            ELSE IF recipesAmount <= 0 THEN
                DISPLAY "Please choose a number above 0"
                WAIT for user
            ELSE
                EXIT LOOP
            END IF
        UNTIL valid input

        RETURN budgetMin, budgetMax, recipesAmount

    END FUNCTION



    FUNCTION FetchPriceData()

        TRY
            LOAD recipes from RECIPES_FILE_PATH
            LOAD ingredientsMapping from INGREDIENTS_MAPPING_FILE_PATH

            TRY
                LOAD measurementsMapping from MEASUREMENTS_MAPPING_FILE_PATH
            CATCH error
                SET measurementsMapping to empty value
            END TRY

            LOAD bilkaPrices from BILKA_PRICES_FILE_PATH
            LOAD nettoPrices from NETTO_PRICES_FILE_PATH
            LOAD foetexPrices from FOETEX_PRICES_FILE_PATH
            LOAD remaPrices from REMA_PRICES_FILE_PATH

            RETURN recipes, ingredientsMapping, measurementsMapping,
                   bilkaPrices, nettoPrices, foetexPrices, remaPrices

        CATCH error
            DISPLAY "Error loading data from file"
            RETURN NULL
        END TRY

    END FUNCTION



    FUNCTION FindCheapestPrice(ingredient, bilka, netto, foetex, rema)

        SET cheapestPrice = 0
        SET store = empty text

        SET priceBilka = bilka[ingredient]
        SET priceNetto = netto[ingredient]
        SET priceFoetex = foetex[ingredient]
        SET priceRema = rema[ingredient]

        IF priceBilka is lower than priceNetto, priceFoetex, and priceRema THEN
            SET store = "bilka"
            SET cheapestPrice = priceBilka

        ELSE IF priceNetto is lower than priceBilka, priceFoetex, and priceRema THEN
            SET store = "netto"
            SET cheapestPrice = priceNetto

        ELSE IF priceFoetex is lower than priceBilka, priceNetto, and priceRema THEN
            SET store = "foetex"
            SET cheapestPrice = priceFoetex

        ELSE IF priceRema is lower than priceBilka, priceNetto, and priceFoetex THEN
            SET store = "rema"
            SET cheapestPrice = priceRema

        ELSE
            RETURN NULL, NULL
        END IF

        RETURN cheapestPrice, store

    END FUNCTION



    FUNCTION RunAlgorithm(amount, budgetMin, budgetMax)

        SET data = FetchPriceData()

        UNPACK data into:
            recipes,
            ingredientsMapping,
            measurementsMapping,
            bilkaPrices,
            nettoPrices,
            foetexPrices,
            remaPrices

        SET results = empty dictionary

        FOR i from 1 to amount

            SET candidates = empty dictionary
            SET cheapestCandidatePrice = very large number
            SET cheapestCandidateName = NULL
            SET cheapestCandidateStores = NULL

            FOR EACH recipe in recipes

                SET name = recipe["name"]

                IF name exists in results THEN
                    CONTINUE to next recipe
                END IF

                SET ingredients = recipe["ingredients"]
                SET ingredientsResult = empty dictionary
                SET totalPrice = 0

                FOR EACH ingredient in ingredients

                    SET ingredientDK = ingredientsMapping[ingredient]

                    SET cheapestPrice, store =
                        FindCheapestPrice(
                            ingredientDK,
                            bilkaPrices,
                            nettoPrices,
                            foetexPrices,
                            remaPrices
                        )

                    SET ingredientsResult[ingredient] =
                        { store: store, price: cheapestPrice }

                    ADD cheapestPrice to totalPrice

                END FOR

                IF cheapestPrice is between budgetMin and budgetMax THEN
                    ADD name to candidates

                    IF totalPrice < cheapestCandidatePrice THEN
                        SET cheapestCandidatePrice = totalPrice
                        SET cheapestCandidateName = name
                        SET cheapestCandidateStores = ingredientsResult
                    END IF
                END IF

            END FOR

            IF cheapestCandidateName is not NULL THEN
                SET results[cheapestCandidateName] =
                    {
                        price: cheapestCandidatePrice,
                        stores: cheapestCandidateStores
                    }
            END IF

        END FOR

        RETURN results

    END FUNCTION



    FUNCTION Main()

        SET budgetMin, budgetMax, recipesAmount = GetInputs()

        TRY
            SET results = RunAlgorithm(recipesAmount, budgetMin, budgetMax)
            DISPLAY results

        CATCH error
            DISPLAY "ERROR:"
            DISPLAY error
            WAIT for user
        END TRY

    END FUNCTION

END
