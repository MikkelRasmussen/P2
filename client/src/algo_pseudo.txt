    price_føtex = netto[ingredient]
    price_rema = netto[ingredient]

    if price_bilka < all(price_netto, price_føtex, price_rema):
        store = "bilka"
        cheapest = price_bilka

    else if price_netto < all(price_bilka, price_føtex, price_rema):
        store = "netto"
        cheapest = price_netto

    else if price_føtex < all(price_netto, price_bilka, price_rema):
        store = "føtex"
        cheapest = price_føtex

    else if price_rema < all(price_netto, price_føtex, price_bilka):
        store = "rema"
        cheapest = price_rema

    else:
        return null, null

    return cheapest, store
}
