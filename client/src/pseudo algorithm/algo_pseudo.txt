BEGIN

    SET file path for recipes
    SET file path for ingredient mapping
    SET file path for measurement mapping
    SET file path for Bilka prices
    SET file path for Netto prices
    SET file path for Foetex prices
    SET file path for Rema prices


    FUNCTION GetInputs

        REPEAT
            ASK user for minimum price per recipe
            READ budgetMin

            IF budgetMin is not a valid number THEN
                DISPLAY error message
                WAIT for user
            ELSE IF budgetMin is less than 0 THEN
                DISPLAY error message
                WAIT for user
            ELSE
                EXIT loop
            END IF
        END REPEAT


        REPEAT
            ASK user for maximum price per recipe
            READ budgetMax

            IF budgetMax is not a valid number THEN
                DISPLAY error message
                WAIT for user
            ELSE IF budgetMax is less than budgetMin THEN
                DISPLAY error message
                WAIT for user
            ELSE IF budgetMax is less than or equal to 0 THEN
                DISPLAY error message
                WAIT for user
            ELSE
                EXIT loop
            END IF
        END REPEAT


        REPEAT
            ASK user for number of recipes
            READ recipesAmount

            IF recipesAmount is not a whole number THEN
                DISPLAY error message
                WAIT for user
            ELSE IF recipesAmount is less than or equal to 0 THEN
                DISPLAY error message
                WAIT for user
            ELSE
                EXIT loop
            END IF
        END REPEAT

        RETURN budgetMin, budgetMax, recipesAmount

    END FUNCTION



    FUNCTION FetchData

        FUNCTION LoadJsonFile(path, defaultValue, required)

            TRY
                OPEN file at path
                READ JSON data
                RETURN JSON data

            IF file does not exist THEN
                IF required is true THEN
                    RAISE error
                ELSE
                    RETURN defaultValue
                END IF

            IF JSON is invalid THEN
                RAISE error
            END TRY

        END FUNCTION


        TRY
            LOAD recipes from recipe file
            LOAD ingredientsMapping from ingredient mapping file
            LOAD measurementsMapping from measurement mapping file if available
            LOAD Bilka prices
            LOAD Netto prices
            LOAD Foetex prices
            LOAD Rema prices

            RETURN a data structure containing:
                recipes
                ingredientsMapping
                measurementsMapping
                prices for all stores

        IF an error occurs THEN
            DISPLAY error message
            WAIT for user
            CLOSE program
        END TRY

    END FUNCTION



    FUNCTION FindCheapestPrice(ingredient, priceData, memoryScores)

        IF ingredient exists in memoryScores THEN
            SET score to memoryScores[ingredient]
        ELSE
            SET score to 1
        END IF

        GET Bilka, Netto, Foetex, and Rema price data from priceData

        SET BilkaPrice to a very large number
        FOR EACH item in Bilka data
            IF ingredient is found in item description THEN
                IF item price is lower than BilkaPrice THEN
                    SET BilkaPrice to item price multiplied by score
                END IF
            END IF
        END FOR

        SET NettoPrice to a very large number
        FOR EACH item in Netto data
            IF ingredient is found in item description THEN
                IF item price is lower than NettoPrice THEN
                    SET NettoPrice to item price multiplied by score
                END IF
            END IF
        END FOR

        SET FoetexPrice to a very large number
        FOR EACH item in Foetex data
            IF ingredient is found in item description THEN
                IF item price is lower than FoetexPrice THEN
                    SET FoetexPrice to item price multiplied by score
                END IF
            END IF
        END FOR

        SET RemaPrice to a very large number
        FOR EACH department in Rema departments
            FOR EACH category in department categories
                FOR EACH item in category items
                    IF ingredient is found in item name THEN
                        IF item price is lower than RemaPrice THEN
                            SET RemaPrice to item price multiplied by score
                        END IF
                    END IF
                END FOR
            END FOR
        END FOR

        CREATE a collection of store prices:
            Bilka
            Netto
            Foetex
            Rema

        FIND the store with the lowest price
        SET cheapestPrice to that lowest price
        SET cheapestStore to that store

        IF cheapestPrice was not found THEN
            RETURN NULL, NULL
        END IF

        RETURN cheapestPrice, cheapestStore

    END FUNCTION



    FUNCTION RunAlgorithm(amount, budgetMin, budgetMax, memoryScores)

        LOAD all data using FetchData

        GET recipes from data
        GET ingredient mapping from data
        GET store prices from data

        SET results to empty collection

        REPEAT amount times

            SET cheapestCandidatePrice to a very large number
            SET cheapestCandidateName to NULL
            SET cheapestCandidateStores to NULL
            SET candidates to empty collection

            FOR EACH recipeRow in recipes

                GET meals from recipeRow

                FOR EACH recipe in meals

                    SET name to recipe meal name

                    IF name already exists in results THEN
                        SKIP this recipe
                    END IF

                    SET ingredients to empty list

                    FOR number from 1 to 20
                        GET ingredient field from recipe
                        CLEAN the ingredient text
                        IF ingredient exists THEN
                            ADD ingredient to ingredients list
                        END IF
                    END FOR

                    SET recipePrices to empty collection
                    SET totalPrice to 0

                    FOR EACH ingredient in ingredients

                        TRANSLATE ingredient to Danish using ingredient mapping

                        FIND cheapest price and cheapest store by calling FindCheapestPrice

                        IF no price or store is found THEN
                            SKIP this ingredient
                        END IF

                        SAVE ingredient price and store in recipePrices
                        ADD cheapest price to totalPrice

                    END FOR

                    IF totalPrice is between budgetMin and budgetMax THEN
                        SAVE recipe in candidates

                        IF totalPrice is lower than cheapestCandidatePrice THEN
                            SET cheapestCandidatePrice to totalPrice
                            SET cheapestCandidateName to name
                            SET cheapestCandidateStores to recipePrices
                        END IF
                    END IF

                END FOR

            END FOR

            IF cheapestCandidateName exists THEN
                ADD recipe to results with:
                    recipe name
                    total price
                    ingredient store details
            END IF

        END REPEAT

        RETURN results

    END FUNCTION



    FUNCTION Main

        GET budgetMin, budgetMax, and recipesAmount from GetInputs
        SET memoryScores to empty collection

        TRY
            RUN algorithm with recipesAmount, budgetMin, budgetMax, and memoryScores
            STORE result in results

        IF an error occurs THEN
            DISPLAY error message
            DISPLAY full error details
            WAIT for user
        END TRY

    END FUNCTION

END
