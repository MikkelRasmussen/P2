BEGIN

    SET recipe file path
    SET ingredient mapping file path
    SET measurement mapping file path
    SET Bilka price file path
    SET Netto price file path
    SET Foetex price file path
    SET Rema1000 price file path


    FUNCTION GetInputs

        REPEAT
            ASK user for minimum price per recipe
            READ budgetMin

            IF budgetMin is not a number THEN
                DISPLAY "Please choose a number"
                WAIT for user to try again
            ELSE IF budgetMin < 0 THEN
                DISPLAY "Please choose a number above 0"
                WAIT for user to try again
            ELSE
                STOP loop
            END IF
        END REPEAT


        REPEAT
            ASK user for maximum price per recipe
            READ budgetMax

            IF budgetMax is not a number THEN
                DISPLAY "Please choose a number"
                WAIT for user to try again
            ELSE IF budgetMax < budgetMin THEN
                DISPLAY "Max price can not be lower than minimum price"
                WAIT for user to try again
            ELSE IF budgetMax <= 0 THEN
                DISPLAY "Please choose a number above 0"
                WAIT for user to try again
            ELSE
                STOP loop
            END IF
        END REPEAT


        REPEAT
            ASK user for amount of recipes
            READ recipesAmount

            IF recipesAmount is not a whole number THEN
                DISPLAY "Please choose a whole number"
                WAIT for user to try again
            ELSE IF recipesAmount <= 0 THEN
                DISPLAY "Please choose a number above 0"
                WAIT for user to try again
            ELSE
                STOP loop
            END IF
        END REPEAT

        RETURN budgetMin, budgetMax, recipesAmount

    END FUNCTION



    FUNCTION FetchPriceData

        TRY
            LOAD recipes
            LOAD ingredientsMapping
            TRY
                LOAD measurementsMapping
            CATCH error
                SET measurementsMapping to empty value
            END TRY

            LOAD bilkaPrices
            LOAD nettoPrices
            LOAD foetexPrices
            LOAD remaPrices

            RETURN recipes, ingredientsMapping, measurementsMapping,
                   bilkaPrices, nettoPrices, foetexPrices, remaPrices

        CATCH error
            DISPLAY "Error loading data from file"
            WAIT for user
            CLOSE program
        END TRY

    END FUNCTION



    FUNCTION FindCheapestPrice(ingredient, bilka, netto, foetex, rema)

        SET cheapestPrice to 0
        SET cheapestStore to empty

        SET priceBilka to price of ingredient in Bilka
        SET priceNetto to price of ingredient in Netto
        SET priceFoetex to price of ingredient in Foetex
        SET priceRema to price of ingredient in Rema

        IF Bilka has the lowest price THEN
            SET cheapestPrice to priceBilka
            SET cheapestStore to "bilka"

        ELSE IF Netto has the lowest price THEN
            SET cheapestPrice to priceNetto
            SET cheapestStore to "netto"

        ELSE IF Foetex has the lowest price THEN
            SET cheapestPrice to priceFoetex
            SET cheapestStore to "foetex"

        ELSE IF Rema has the lowest price THEN
            SET cheapestPrice to priceRema
            SET cheapestStore to "rema"

        ELSE
            RETURN NULL, NULL
        END IF

        RETURN cheapestPrice, cheapestStore

    END FUNCTION



    FUNCTION RunAlgorithm(amount, budgetMin, budgetMax)

        LOAD all recipe, mapping, and store price data by calling FetchPriceData

        SET results to empty collection

        FOR each number from 1 to amount

            SET cheapestCandidatePrice to very large number
            SET cheapestCandidateName to NULL
            SET cheapestCandidateStores to NULL

            FOR EACH recipe in recipes

                SET name to recipe name

                IF name is already in results THEN
                    SKIP this recipe
                END IF

                SET ingredients to recipe ingredients
                SET ingredientsResult to empty collection
                SET totalPrice to 0

                FOR EACH ingredient in ingredients

                    SET ingredientDK to translated ingredient from ingredientsMapping

                    SET cheapestPrice and store by calling FindCheapestPrice

                    IF cheapestPrice is NULL OR store is NULL THEN
                        SKIP this ingredient
                    END IF

                    SAVE ingredient with store and price in ingredientsResult
                    ADD cheapestPrice to totalPrice

                END FOR

                IF cheapestPrice is between budgetMin and budgetMax THEN

                    IF totalPrice is lower than cheapestCandidatePrice THEN
                        SET cheapestCandidatePrice to totalPrice
                        SET cheapestCandidateName to name
                        SET cheapestCandidateStores to ingredientsResult
                    END IF

                END IF

            END FOR

            IF cheapestCandidateName exists THEN
                SAVE cheapestCandidateName, cheapestCandidatePrice,
                     and cheapestCandidateStores in results
            END IF

        END FOR

        RETURN results

    END FUNCTION



    FUNCTION Main

        GET budgetMin, budgetMax, and recipesAmount from GetInputs

        TRY
            SET results to RunAlgorithm(recipesAmount, budgetMin, budgetMax)
            DISPLAY results
        CATCH error
            DISPLAY "ERROR"
            DISPLAY error
            WAIT for user
        END TRY

    END FUNCTION

END
