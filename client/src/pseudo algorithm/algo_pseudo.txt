BEGIN

    SET recipe file path
    SET ingredient mapping file path
    SET measurement mapping file path
    SET Bilka price file path
    SET Netto price file path
    SET Foetex price file path
    SET Rema1000 price file path


    FUNCTION GetInputs

        REPEAT
            ASK user for minimum price per recipe
            READ budgetMin

            IF budgetMin is not a number THEN
                DISPLAY "Please choose a number"
                WAIT for user to try again

            ELSE IF budgetMin is less than 0 THEN
                DISPLAY "Please choose a number above 0"
                WAIT for user to try again

            ELSE
                STOP loop
            END IF
        END REPEAT


        REPEAT
            ASK user for maximum price per recipe
            READ budgetMax

            IF budgetMax is not a number THEN
                DISPLAY "Please choose a number"
                WAIT for user to try again

            ELSE IF budgetMax is less than budgetMin THEN
                DISPLAY "Max price can not be lower than minimum price"
                WAIT for user to try again

            ELSE IF budgetMax is less than or equal to 0 THEN
                DISPLAY "Please choose a number above 0"
                WAIT for user to try again

            ELSE
                STOP loop
            END IF
        END REPEAT


        REPEAT
            ASK user for number of recipes
            READ recipesAmount

            IF recipesAmount is not a whole number THEN
                DISPLAY "Please choose a whole number"
                WAIT for user to try again

            ELSE IF recipesAmount is less than or equal to 0 THEN
                DISPLAY "Please choose a number above 0"
                WAIT for user to try again

            ELSE
                STOP loop
            END IF
        END REPEAT

        RETURN budgetMin, budgetMax, recipesAmount

    END FUNCTION



    FUNCTION FetchPriceData

        TRY
            LOAD recipes
            LOAD ingredientsMapping

            TRY
                LOAD measurementsMapping
            CATCH error
                SET measurementsMapping to empty value
            END TRY

            LOAD bilkaPrices
            LOAD nettoPrices
            LOAD foetexPrices
            LOAD remaPrices

            RETURN recipes, ingredientsMapping, measurementsMapping,
                   bilkaPrices, nettoPrices, foetexPrices, remaPrices

        CATCH error
            DISPLAY "Error loading data from file"
            WAIT for user
            CLOSE program
        END TRY

    END FUNCTION



    FUNCTION FindCheapestPrice(ingredient, bilka, netto, foetex, rema)

        SET priceBilka to a very large number
        FOR EACH item in bilka
            IF ingredient is found in item description THEN
                IF item price is lower than priceBilka THEN
                    SET priceBilka to item price
                END IF
            END IF
        END FOR


        SET priceNetto to a very large number
        FOR EACH item in netto
            IF ingredient is found in item description THEN
                IF item price is lower than priceNetto THEN
                    SET priceNetto to item price
                END IF
            END IF
        END FOR


        SET priceFoetex to a very large number
        FOR EACH item in foetex
            IF ingredient is found in item description THEN
                IF item price is lower than priceFoetex THEN
                    SET priceFoetex to item price
                END IF
            END IF
        END FOR


        SET priceRema to a very large number
        FOR EACH department in rema departments
            FOR EACH category in department categories
                FOR EACH item in category items
                    IF ingredient is found in item name THEN
                        IF item pricing price is lower than priceRema THEN
                            SET priceRema to item pricing price
                        END IF
                    END IF
                END FOR
            END FOR
        END FOR


        CREATE prices with:
            bilka = priceBilka
            netto = priceNetto
            foetex = priceFoetex
            rema = priceRema

        FIND the store with the lowest price
        SET cheapest to the lowest price
        SET store to the matching store name

        IF cheapest is still the very large number THEN
            RETURN NULL, NULL
        END IF

        RETURN cheapest, store

    END FUNCTION



    FUNCTION RunAlgorithm(amount, budgetMin, budgetMax)

        LOAD all data by calling FetchPriceData

        SET results to empty collection

        FOR each number from 1 to amount

            SET cheapestCandidatePrice to a very large number
            SET cheapestCandidateName to NULL
            SET cheapestCandidateStores to NULL

            FOR EACH recipe in recipes

                SET name to recipe name

                IF name already exists in results THEN
                    SKIP this recipe
                END IF

                SET ingredients to recipe ingredients
                SET ingredientsResult to empty collection
                SET totalPrice to 0

                FOR EACH ingredient in ingredients

                    SET ingredientDK to translated ingredient from ingredientsMapping

                    CALL FindCheapestPrice with ingredientDK and all store price data
                    RECEIVE cheapestPrice and store

                    IF cheapestPrice is NULL OR store is NULL THEN
                        SKIP this ingredient
                    END IF

                    SAVE ingredient with its store and price in ingredientsResult
                    ADD cheapestPrice to totalPrice

                END FOR

                IF cheapestPrice is between budgetMin and budgetMax THEN
                    IF totalPrice is lower than cheapestCandidatePrice THEN
                        SET cheapestCandidatePrice to totalPrice
                        SET cheapestCandidateName to name
                        SET cheapestCandidateStores to ingredientsResult
                    END IF
                END IF

            END FOR

            IF cheapestCandidateName exists THEN
                SAVE in results:
                    recipe name
                    total price
                    ingredient store details
            END IF

        END FOR

        RETURN results

    END FUNCTION



    FUNCTION Main

        CALL GetInputs
        RECEIVE budgetMin, budgetMax, recipesAmount

        TRY
            CALL RunAlgorithm with recipesAmount, budgetMin, and budgetMax
            DISPLAY results

        CATCH error
            DISPLAY "ERROR"
            DISPLAY error
            WAIT for user
        END TRY

    END FUNCTION

END
