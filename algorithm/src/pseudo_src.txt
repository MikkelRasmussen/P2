    FUNCTION CalculateBuyPrice(recipe, store, currentItem)

        SET lists of known volume units
        SET lists of known weight units
        SET lists of miscellaneous units
        SET lists of abstract units
        SET lists of ignored units

        GET recipe quantity from recipe
        GET recipe unit from recipe
        CLEAN recipe unit

        GET store price from store
        GET store unit from store
        CLEAN store unit

        SET resultingPrice to very large number
        SET amountToBuy to 0
        SET storeQuantity to null
        SET storeQuantityUnit to empty text
        SET packagePrice to store price

        CONVERT recipe quantity to number
        CONVERT store price to number

        IF recipe quantity or store price is not a number THEN
            RETURN resultingPrice and amountToBuy
        END IF

        NORMALISE recipe unit:
            grams become g
            kilograms become kg
            milliliters become ml
            liters become l
            tablespoons become tbs
            teaspoons become tsp
            empty unit becomes stk
            miscellaneous, abstract, or ignored units become stk

        NORMALISE store unit:
            grams become g
            kilograms become kg
            milliliters become ml
            liters become l
            tablespoons become tbs
            teaspoons become tsp
            empty unit becomes stk

        IF currentItem exists THEN

            TRY to read package price from currentItem price
            IF package price is not valid THEN
                USE store price as package price
            END IF

            TRY to read package quantity from currentItem contents

            IF package quantity is valid THEN
                READ and clean package quantity unit from currentItem contentsUnit
            ELSE
                SET package quantity to null
            END IF

        END IF

        NORMALISE package quantity unit:
            grams become g
            kilograms become kg
            milliliters become ml
            liters become l
            miscellaneous, abstract, ignored, or empty units become stk

        SET weight conversion values to grams
        SET volume conversion values to milliliters

        IF package quantity exists and is greater than 0 THEN

            IF recipe unit equals package quantity unit THEN
                SET amountToBuy to recipe quantity divided by package quantity, rounded up

            ELSE IF recipe unit is a weight unit and package unit can be converted to grams THEN
                CONVERT recipe quantity to grams
                CONVERT package quantity to grams
                SET amountToBuy to recipe grams divided by package grams, rounded up

            ELSE IF recipe unit is a volume unit and package unit can be converted to milliliters THEN
                CONVERT recipe quantity to milliliters
                CONVERT package quantity to milliliters
                SET amountToBuy to recipe milliliters divided by package milliliters, rounded up

            ELSE IF both recipe unit and package unit are stk THEN
                SET amountToBuy to recipe quantity divided by package quantity, rounded up
            END IF

            IF amountToBuy is greater than 0 THEN
                SET resultingPrice to packagePrice multiplied by amountToBuy
            END IF

        END IF

        IF resultingPrice is still the very large number THEN

            IF recipe unit equals store unit THEN
                SET resultingPrice to store price multiplied by recipe quantity
                SET amountToBuy to recipe quantity rounded up

            ELSE IF recipe unit is a weight unit and store unit can be converted to grams THEN
                CONVERT recipe quantity to grams
                CONVERT store unit to grams
                SET resultingPrice to store price multiplied by recipe grams divided by store grams
                SET amountToBuy to 1

            ELSE IF recipe unit is a volume unit and store unit can be converted to milliliters THEN
                CONVERT recipe quantity to milliliters
                CONVERT store unit to milliliters
                SET resultingPrice to store price multiplied by recipe milliliters divided by store milliliters
                SET amountToBuy to 1

            ELSE IF both recipe unit and store unit are stk THEN
                SET resultingPrice to store price multiplied by recipe quantity
                SET amountToBuy to recipe quantity rounded up
            END IF

        END IF

        IF resultingPrice is not the very large number THEN
            ROUND resultingPrice to 2 decimals
        END IF

        RETURN resultingPrice and amountToBuy

    END FUNCTION



    FUNCTION FindCheapestPrice(ingredient, quantity, unit, priceData)

        TRY
            GET Bilka data from priceData
            GET Netto data from priceData
            GET Foetex data from priceData
        IF an error occurs THEN
            RETURN null, null, null, false
        END TRY

        CREATE recipeQuantity object with:
            quantity
            unit

        SET ingredientLower to lowercase ingredient


        SET Bilka best price to very large number
        SET Bilka amount to buy to 0
        SET Bilka product name to null
        SET Bilka unit price to null
        SET Bilka unit to null

        FOR EACH item in Bilka data

            IF ingredientLower is found in item description THEN

                GET item unit price
                GET item price unit

                CREATE storeQuantity object with:
                    price as item unit price
                    unit as item price unit

                CALL CalculateBuyPrice using recipeQuantity, storeQuantity, and current item

                IF calculated price is lower than Bilka best price THEN
                    SAVE calculated price
                    SAVE amount to buy
                    SAVE unit price
                    SAVE unit
                    SAVE product name
                END IF

            END IF

        END FOR


        REPEAT the same process for Netto data


        REPEAT the same process for Foetex data


        CREATE prices object containing:
            Bilka calculated price
            Netto calculated price
            Foetex calculated price

        SET store to Bilka by default

        IF Netto price is lower than current selected store price THEN
            SET store to Netto
        END IF

        IF Foetex price is lower than current selected store price THEN
            SET store to Foetex
        END IF

        SET cheapestPrice to prices[store]

        CREATE storesData object containing data for each store:
            price
            amount to buy
            product name
            price per unit
            unit

        IF cheapestPrice is still the very large number THEN
            RETURN null, null, null, false
        END IF

        RETURN cheapestPrice, store, storesData, true

    END FUNCTION



    FUNCTION RunAlgorithm(amount, budgetMin, budgetMax, memoryScores)

        LOAD all data using FetchData

        GET recipes from data
        GET ingredient mapping from data
        GET store prices from data

        SET basicIngredients to list of ingredients that should be skipped


        FUNCTION GetMemoryScore(name)

            TRY
                IF name exists in memoryScores THEN
                    RETURN memoryScores[name] as number

                ELSE IF lowercase name exists in memoryScores THEN
                    RETURN memoryScores[lowercase name] as number

                ELSE
                    RETURN 1
                END IF

            IF an error occurs THEN
                RETURN 1
            END TRY

        END FUNCTION


        FUNCTION ParseQuantity(quantity)

            CONVERT quantity to lowercase text and trim whitespace

            REPLACE fraction symbols with normal fraction text:
                ½ becomes 1/2
                ¼ becomes 1/4
                ¾ becomes 3/4
                ⅓ becomes 1/3
                ⅔ becomes 2/3
                ⅛ becomes 1/8

            REPLACE comma with dot
            REPLACE dash with space
            REPLACE opening parenthesis with space
            REPLACE closing parenthesis with space


            FUNCTION ReadNumber(number)

                TRIM number

                IF number contains space and slash THEN
                    SPLIT number into whole number and fraction
                    CONVERT whole number to number
                    SPLIT fraction into numerator and denominator
                    RETURN whole number plus numerator divided by denominator
                END IF

                IF number contains slash THEN
                    SPLIT number into numerator and denominator
                    RETURN numerator divided by denominator
                END IF

                RETURN number converted to numeric value

            END FUNCTION


            FUNCTION IsDigit(character)

                RETURN true if character is between 0 and 9
                OTHERWISE return false

            END FUNCTION


            FUNCTION ReadNumberText(text, startPosition)

                IF startPosition is outside text OR character at startPosition is not a digit THEN
                    RETURN empty text and startPosition
                END IF

                SET start to startPosition
                SET position to startPosition

                WHILE position is inside text AND character at position is a digit
                    INCREASE position by 1
                END WHILE

                SET afterWholeNumber to position

                CHECK for mixed fraction number, for example 1 1/2:
                    SKIP spaces after whole number
                    READ fraction numerator
                    IF there was space and a slash follows numerator THEN
                        READ denominator
                        IF denominator exists THEN
                            RETURN number text and end position
                        END IF
                    END IF

                CHECK for normal fraction, for example 1/2:
                    IF slash follows first number THEN
                        READ denominator
                        IF denominator exists THEN
                            RETURN fraction text and end position
                        END IF
                    END IF

                CHECK for decimal number, for example 1.5:
                    IF dot follows first number and a digit follows dot THEN
                        READ digits after dot
                    END IF

                RETURN number text and end position

            END FUNCTION


            FUNCTION ReadUnitText(text, startPosition)

                SET position to startPosition

                SKIP spaces

                SET unit to empty text

                WHILE position is inside text AND character is a lowercase letter
                    ADD character to unit
                    INCREASE position by 1
                END WHILE

                IF unit exists THEN
                    RETURN unit
                END IF

                RETURN stk

            END FUNCTION


            CHECK if measurement has format "number x number unit":
                SCAN through quantity text
                TRY to read first number
                IF first number exists THEN
                    SKIP spaces
                    IF next character is x THEN
                        SKIP spaces after x
                        TRY to read second number
                        IF second number exists THEN
                            SET amount to first number multiplied by second number
                            READ unit after second number
                            RETURN amount and unit
                        END IF
                    END IF
                END IF

            CHECK for first normal number:
                SCAN through quantity text
                TRY to read number
                IF number exists THEN
                    CONVERT number text to numeric amount
                    READ unit after number
                    RETURN amount and unit
                END IF

            IF no number exists THEN
                SPLIT quantity into words
                IF at least one word exists THEN
                    RETURN 1 and first word
                END IF
            END IF

            RETURN 1 and stk

        END FUNCTION



        SET results to empty object
        SET recipesNotPriced to empty object

        REPEAT amount times

            SET candidates to empty object
            SET cheapestCandidatePrice to very large number
            SET cheapestCandidateScore to very large number
            SET cheapestCandidateName to null
            SET cheapestCandidateStores to null

            FOR EACH recipeRow in recipes

                GET meals from recipeRow
                IF meals does not exist THEN
                    USE empty list

                FOR EACH recipe in meals

                    GET recipe name

                    IF recipe name already exists in results THEN
                        SKIP this recipe
                    END IF

                    SET ingredientsToValue to empty list

                    FOR n from 1 to 20

                        GET ingredient field strIngredient n
                        GET measure field strMeasure n

                        IF ingredient exists THEN
                            TRIM ingredient
                        END IF

                        IF measure exists THEN
                            TRIM measure
                        END IF

                        IF ingredient exists after trimming THEN
                            SET normalizedIngredient to lowercase ingredient

                            IF normalizedIngredient is not in basicIngredients THEN
                                ADD object to ingredientsToValue:
                                    name as ingredient
                                    measure as measure or empty text
                            END IF
                        END IF

                    END FOR

                    IF ingredientsToValue is empty THEN
                        MARK recipe as not priced
                        SKIP this recipe
                    END IF

                    SET recipePrices to empty object
                    SET totalPrice to 0
                    SET rankingPrice to 0
                    SET successBool to true

                    FOR EACH ingredientData in ingredientsToValue

                        GET ingredient name
                        GET ingredient measurement

                        TRY
                            PARSE measurement using ParseQuantity
                            GET quantity and unit
                        IF an error occurs THEN
                            SET successBool to false
                            STOP ingredient loop
                        END TRY

                        TRANSLATE ingredient to Danish using ingredientsMapping

                        IF Danish ingredient was not found THEN
                            SET successBool to false
                            STOP ingredient loop
                        END IF

                        TRY
                            CALL FindCheapestPrice with Danish ingredient, quantity, unit, and priceData
                            GET cheapestPrice, store, storesData, and found
                        IF an error occurs THEN
                            DISPLAY error message
                            SET successBool to false
                            STOP ingredient loop
                        END TRY

                        IF found is false OR cheapestPrice is null OR store is null OR storesData is null THEN
                            SET successBool to false
                            STOP ingredient loop
                        END IF

                        GET ingredientScore using original ingredient name

                        IF ingredientScore equals 1 THEN
                            GET ingredientScore using Danish ingredient name
                        END IF

                        SAVE ingredient data in recipePrices:
                            measure
                            selected store
                            storesData
                            selected product name
                            amount to buy
                            cheapest price
                            memory score

                        ADD cheapestPrice to totalPrice
                        ADD cheapestPrice multiplied by ingredientScore to rankingPrice

                    END FOR

                    IF successBool is false THEN
                        MARK recipe as not priced
                        SKIP this recipe
                    END IF

                    IF totalPrice is between budgetMin and budgetMax THEN

                        SAVE recipePrices in candidates using recipe name

                        IF rankingPrice is lower than cheapestCandidateScore THEN
                            SET cheapestCandidatePrice to totalPrice
                            SET cheapestCandidateScore to rankingPrice
                            SET cheapestCandidateName to recipe name
                            SET cheapestCandidateStores to recipePrices
                        END IF

                    END IF

                END FOR

            END FOR

            IF cheapestCandidateName exists THEN
                ADD recipe to results with:
                    price
                    score
                    stores
            END IF

        END REPEAT

        SET notPricedCount in results to number of keys in recipesNotPriced

        RETURN results

    END FUNCTION
